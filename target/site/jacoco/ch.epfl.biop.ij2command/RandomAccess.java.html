package ch.epfl.biop.ij2command;

import java.util.Vector;

import ij.IJ;

public class RandomAccess {
	int length;
	Vector<Integer>list=new Vector<Integer>();

	RandomAccess(int[] array){
		this.length=array.length;
		for (int i=0;i<length;i++) {
			list.add(array[i]);
		}
		
	}
	RandomAccess(Vector<Integer>input){
		this.list=input;
		this.length=list.size();
	}
	int [] getList() {
		int [] outArray=new int [length];
		
		for (int i=length;i>0;i--) {
			int order=(int)Math.round(Math.random()*(i-1));
			outArray[i-1]=list.get(order);
			//IJ.log("i="+i+"     order"+order+"    "+outArray[i-1]+"   "+list.get(order));
			list.remove(order);
			
		}
		
		return outArray;
	}
	
}
